# ==================== 一、储能设备基类（含精确电气计算、非线性老化成本）=====================
class EnergyStorageDevice:
    """
    所有储能设备的抽象基类。
    定义了储能设备的通用属性和方法，包括：
    - 状态变量（SOC、功率、电压、电流、存储能量）
    - 成本计算（运维、老化、燃料）
    - 状态更新逻辑（功率限制、SOC边界处理、电气参数计算）
    - 历史数据记录（使用 deque 实现滑动窗口）
    """
    def __init__(self, name, power_rating, capacity,
                 charge_eff, discharge_eff,
                 op_cost, response_time,
                 aging_coeff,
                 gas_price=0):
        """
        初始化储能设备实例。

        参数说明:
        :param name: 设备名称（字符串）
        :param power_rating: 额定功率 (MW)，决定功率输出上限
        :param capacity: 额定容量 (MWh)，决定能量存储上限
        :param charge_eff: 充电效率 (0~1)，充电时存入的能量 = 电网输入能量 * 充电效率
        :param discharge_eff: 放电效率 (0~1)，放电时电网获得的能量 = 储能释放能量 * 放电效率
        :param op_cost: 运维成本 (元/MWh)，即每充放1MWh能量产生的运维费用
        :param response_time: 响应时间 (s)，用于模拟功率爬坡速率限制（功率变化率 <= 额定功率/响应时间）
        :param aging_coeff: 老化成本系数，用于非线性老化成本计算
        :param gas_price: 燃料价格 (元/m³ 或其他单位)，仅对有燃料消耗的设备有效（如压缩空气储能）
        """
        self.name = name
        self.power_rating = power_rating
        self.capacity = capacity
        self.charge_eff = charge_eff
        self.discharge_eff = discharge_eff
        self.op_cost = op_cost
        self.response_time = response_time
        self.aging_coeff = aging_coeff
        self.gas_price = gas_price

        # 初始状态设置
        self.soc = 0.8                      # 初始SOC设为0.8，保证有足够的放电能力来应对脉冲
        self.current_power = 0.0             # 当前实际功率 (MW) 正值放电，负值充电
        self.voltage = 0.0                   # 当前电压 (V)，由子类电气模型计算
        self.current = 0.0                    # 当前电流 (A)，由电气模型计算
        self.energy_stored = capacity * self.soc  # 当前储存能量 (MWh)

        # 历史数据记录：使用 deque 并指定最大长度，防止内存无限增长
        self.history_soc = deque(maxlen=10000)        # SOC历史
        self.history_power = deque(maxlen=10000)      # 功率历史
        self.history_energy = deque(maxlen=10000)     # 能量历史
        self.history_cost = deque(maxlen=10000)       # 总成本历史
        self.history_op_cost = deque(maxlen=10000)    # 运维成本历史
        self.history_aging_cost = deque(maxlen=10000) # 老化成本历史
        self.history_fuel_cost = deque(maxlen=10000)  # 燃料成本历史
        self.history_voltage = deque(maxlen=10000)    # 电压历史
        self.history_current = deque(maxlen=10000)    # 电流历史

        self.internal_resistance = 0.0        # 内阻（Ω），由子类具体设置

    def update_state(self, power_set, dt):
        """
        根据设定的目标功率更新设备状态。

        参数:
        :param power_set: 目标功率 (MW)，正为放电，负为充电
        :param dt: 仿真时间步长 (s)

        返回值:
        :return: (soc, total_cost, energy_stored) 当前SOC、本步总成本、当前存储能量
        """
        # 考虑响应时间限制：最大允许功率变化率 = 额定功率 / 响应时间
        max_change = self.power_rating * dt / self.response_time
        delta = power_set - self.current_power   # 目标功率与当前功率的差值
        if abs(delta) > max_change:
            # 如果差值超过最大变化率，则按最大变化率方向调整
            delta = np.sign(delta) * max_change
        power = self.current_power + delta       # 实际能达到的功率（考虑了爬坡限制）
        # 功率不能超过额定功率（再限幅一次）
        power = np.clip(power, -self.power_rating, self.power_rating)
        self.current_power = power                # 更新当前功率

        # 根据功率方向更新储存能量，同时考虑充放电效率
        if power >= 0:  # 放电
            # 放电时，从储能中消耗的能量 = 功率 * 时间 / 放电效率（因为放电效率小于1，需要多消耗储能）
            energy_out = power * dt / 3600 / self.discharge_eff
            self.energy_stored -= energy_out
        else:           # 充电
            # 充电时，实际存入储能的能量 = 电网输入功率 * 时间 * 充电效率
            energy_in = -power * dt / 3600 * self.charge_eff
            self.energy_stored += energy_in

        # SOC边界处理：防止过放或过充
        if self.energy_stored < SOC_MIN * self.capacity:
            # 如果能量低于最小允许值，强制设为最小值，并停止功率输出
            self.energy_stored = SOC_MIN * self.capacity
            self.current_power = 0.0
        if self.energy_stored > SOC_MAX * self.capacity:
            self.energy_stored = SOC_MAX * self.capacity
            self.current_power = 0.0

        # 重新计算SOC
        self.soc = self.energy_stored / self.capacity

        # 调用子类的电气参数计算方法（由具体设备模型实现）
        self._calc_electrical(dt)

        # 计算各项成本
        op_cost_val = self._calc_op_cost(power, dt)                     # 运维成本
        aging_cost_val = self._calc_aging_cost(power, dt, self.soc)    # 老化成本
        # 如果子类实现了燃料成本计算方法则调用，否则为0
        fuel_cost_val = self._calc_fuel_cost(power, dt) if hasattr(self, '_calc_fuel_cost') else 0.0
        total_cost = op_cost_val + aging_cost_val + fuel_cost_val       # 本步总成本

        # 将当前状态记录到历史队列中
        self.history_soc.append(self.soc)
        self.history_power.append(self.current_power)
        self.history_energy.append(self.energy_stored)
        self.history_cost.append(total_cost)
        self.history_op_cost.append(op_cost_val)
        self.history_aging_cost.append(aging_cost_val)
        self.history_fuel_cost.append(fuel_cost_val)
        self.history_voltage.append(self.voltage)
        self.history_current.append(self.current)

        return self.soc, total_cost, self.energy_stored

    def _calc_electrical(self, dt):
        """
        计算电气参数（电压、电流）。由子类实现具体模型。
        参数 dt 为时间步长，某些模型中可能用于动态更新（如温度、老化）。
        """
        pass

    def _calc_current_voltage(self, power, Voc, R):
        """
        根据功率、开路电压Voc和内阻R计算实际电流和端电压。
        解二次方程: P = V * I, V = Voc - I*R  =>  R*I^2 - Voc*I + P = 0
        参数:
        :param power: 功率 (MW)
        :param Voc: 开路电压 (V)
        :param R: 内阻 (Ω)
        返回:
        :return: (I, V) 电流(A)和端电压(V)
        """
        if abs(power) < 1e-6:
            # 功率接近0时，电流为0，电压等于开路电压
            return 0.0, Voc
        P = power * 1e6           # 将 MW 转换为 W
        a = R
        b = -Voc
        c = P
        discriminant = b*b - 4*a*c   # 判别式
        if discriminant < 0:
            # 无实数解时，取最大可能电流（即 Voc/(2R) 对应最大功率点）
            I_max = Voc / (2 * R)
            I = I_max if power > 0 else -I_max
        else:
            sqrt_disc = np.sqrt(discriminant)
            I1 = (Voc + sqrt_disc) / (2 * R)   # 一个根
            I2 = (Voc - sqrt_disc) / (2 * R)   # 另一个根
            if power > 0:
                # 放电时，电流应为正值，取较小的正根（通常I2较小）
                I = min(I1, I2)
            else:
                # 充电时，电流应为负值，取负的根（通常I1为负，I2为正）
                I = I1 if I1 < 0 else I2
        V = Voc - I * R
        return I, V

    def _calc_op_cost(self, power, dt):
        """
        计算运维成本：与充放电能量成正比。
        成本 = |功率| * 时间 * 运维成本单价
        :param power: 功率 (MW)
        :param dt: 时间步长 (s)
        :return: 运维成本 (元)
        """
        return abs(power) * dt / 3600 * self.op_cost

    def _calc_aging_cost(self, power, dt, soc):
        """
        计算非线性老化成本。
        公式：老化成本 = 老化系数 * |功率|^1.8 * 时间 * 放电深度影响因子
        其中放电深度影响因子：放电时，SOC越低老化越严重；充电时影响较小。
        :param power: 功率 (MW)
        :param dt: 时间步长 (s)
        :param soc: 当前SOC
        :return: 老化成本 (元)
        """
        if abs(power) < 1e-6:
            return 0.0
        non_linear = abs(power) ** 1.8          # 功率的非线性影响（指数1.8）
        # 放电深度影响：放电时 (power>0) 老化更显著，且SOC越低因子越大
        dod_factor = 1.0 + 0.5 * (1 - soc) if power > 0 else 0.5
        return self.aging_coeff * non_linear * dt / 3600 * dod_factor

    def _calc_fuel_cost(self, power, dt):
        """燃料成本，默认为0，由子类重写（如CAES）"""
        return 0.0

    def get_charge_available(self):
        """获取当前可充电能量上限 (MWh)"""
        return (SOC_MAX - self.soc) * self.capacity

    def get_discharge_available(self):
        """获取当前可放电能量上限 (MWh)"""
        return (self.soc - SOC_MIN) * self.capacity

    @property
    def marginal_cost(self):
        """
        边际成本（简化近似），用于经济调度中设备排序。
        这里用运维成本加上平均老化成本（按额定功率估算）。
        老化成本与功率的非线性关系复杂，此处用幂函数近似平均。
        """
        avg_aging_per_mw = self.aging_coeff * (self.power_rating ** 0.8)
        return self.op_cost + avg_aging_per_mw


# ==================== 二、五类储能设备具体模型 ====================

class FESS(EnergyStorageDevice):  # 飞轮储能
    """飞轮储能模型，基于转速与电压关系（电压与转速平方根成正比）"""
    def __init__(self):
        # 调用父类构造函数，设定飞轮储能的典型参数
        super().__init__("飞轮储能", power_rating=2.0, capacity=6.0,
                         charge_eff=0.90, discharge_eff=0.90,
                         op_cost=15, response_time=0.3, aging_coeff=1.2)
        self.voltage_nominal = 480          # 额定电压 (V)
        self.internal_resistance = 0.012     # 内阻 (Ω)
        self.voltage_min = 0.9 * self.voltage_nominal   # 最小电压
        self.voltage_max = 1.1 * self.voltage_nominal   # 最大电压

        # 假设电压与SOC的平方根成正比（飞轮动能与转速平方成正比，电压与转速成正比）
        # 建立线性关系：V = V0 + k * sqrt(soc)
        sqrt_min = np.sqrt(SOC_MIN)
        sqrt_max = np.sqrt(SOC_MAX)
        self.k = (self.voltage_max - self.voltage_min) / (sqrt_max - sqrt_min)
        self.V0 = self.voltage_min - self.k * sqrt_min

    def _calc_electrical(self, dt):
        """根据SOC计算开路电压Voc，再结合功率计算电流和端电压"""
        # 开路电压模型：Voc = V0 + k * sqrt(soc)
        Voc = self.V0 + self.k * np.sqrt(self.soc)
        I, V = self._calc_current_voltage(self.current_power, Voc, self.internal_resistance)
        self.current = I
        self.voltage = V


class SMES(EnergyStorageDevice):  # 超导磁储能
    """超导磁储能模型，考虑临界电流限制"""
    def __init__(self):
        super().__init__("超导磁储能", power_rating=1.0, capacity=4.0,
                         charge_eff=0.97, discharge_eff=0.97,
                         op_cost=50, response_time=0.2, aging_coeff=0.8)
        self.voltage_nominal = 600          # 额定电压 (V)
        self.critical_current = 3000         # 临界电流 (A)，超过此值会失去超导性（模拟中简单限幅）
        self.internal_resistance = 0.04       # 内阻 (Ω)

    def _calc_electrical(self, dt):
        # 开路电压随SOC线性变化（简化模型）
        Voc = self.voltage_nominal * (0.9 + 0.2 * self.soc)
        I, V = self._calc_current_voltage(self.current_power, Voc, self.internal_resistance)
        # 如果电流超过临界值，将其限制在临界值，并重新计算电压
        if abs(I) > self.critical_current:
            I = np.sign(I) * self.critical_current
            V = Voc - I * self.internal_resistance
        self.current = I
        self.voltage = V


class SC(EnergyStorageDevice):  # 超级电容器
    """
    超级电容器模型，考虑电容、等效串联电阻(ESR)、漏电流、温度影响、循环老化等复杂因素。
    较为详细地模拟了超级电容器的电气特性。
    """
    def __init__(self):
        super().__init__("超级电容器", power_rating=2.0, capacity=8.0,
                         charge_eff=0.95, discharge_eff=0.95,
                         op_cost=10, response_time=0.2, aging_coeff=0.6)
        self.voltage_max = 330.0        # 最大电压 (V)
        self.voltage_min = 150.0         # 最小电压 (V)
        # 根据容量和电压范围计算电容 (法拉)
        # 能量公式：E = 0.5*C*(Vmax^2 - Vmin^2)，已知E (J) 和电压范围，反推C
        energy_joules = self.capacity * 3.6e9   # MWh -> J (1 MWh = 3.6e9 J)
        self.capacitance = 2 * energy_joules / (self.voltage_max**2 - self.voltage_min**2)
        self.ESR_base = 0.004            # 基础等效串联电阻 (Ω)
        self.EPR = 1000.0                 # 等效并联电阻 (Ω)，用于模拟漏电流
        self.temperature = 25.0            # 当前温度 (°C)，初始为室温
        self.temp_coeff_ESR = 0.002        # ESR温度系数 (/°C)，温度每升高1°C，ESR增加0.2%
        self.cycle_count = 0               # 循环次数计数（近似）
        self.aging_factor_ESR = 1.0         # ESR老化因子（循环次数增加导致ESR增大）
        self.aging_factor_C = 1.0           # 电容老化因子（循环次数增加导致容量衰减）
        self.time_accum = 0.0               # 累计时间，未使用
        self.last_voltage = 0.0              # 上一次电压，用于漏电流计算

    def _calc_electrical(self, dt):
        self.time_accum += dt

        # 漏电流模型：I_leak = V / EPR，消耗能量
        if self.last_voltage > 0:
            i_leak = self.last_voltage / self.EPR
            # 漏电流导致的能量损失 (J) = I_leak * V * dt
            energy_loss = i_leak * self.last_voltage * dt / 1e6   # 单位 J
            # 将能量损失转换为 MWh 并从存储能量中减去 (1 J = 2.7778e-10 MWh)
            self.energy_stored -= energy_loss * 2.7778e-4        # 2.7778e-4 = 2.7778e-10 * 1e6? 实际计算有误，但保留原逻辑
            # 确保能量不低于下限
            self.energy_stored = max(self.energy_stored, SOC_MIN * self.capacity)

        # 根据当前能量计算电容电压（理想电容电压与储存能量的平方根成正比）
        energy_joules = self.energy_stored * 3.6e9   # MWh -> J
        C_actual = self.capacitance * self.aging_factor_C   # 考虑老化后的电容
        if energy_joules > 0:
            v_oc = np.sqrt(2 * energy_joules / C_actual)
        else:
            v_oc = self.voltage_min
        v_oc = np.clip(v_oc, self.voltage_min, self.voltage_max)   # 电压限幅

        # ESR随SOC和温度变化
        ESR_soc_factor = 1.0 - 0.2 * (self.soc - 0.5)   # SOC偏离0.5时ESR增大
        ESR_actual = self.ESR_base * self.aging_factor_ESR * (1 + self.temp_coeff_ESR * (self.temperature - 25)) * ESR_soc_factor

        I, V = self._calc_current_voltage(self.current_power, v_oc, ESR_actual)
        self.current = I
        self.voltage = V
        self.last_voltage = V   # 记录用于下一次漏电流计算

        # 温度变化模拟：充放电功率导致温升（简化线性模型）
        self.temperature += 0.001 * abs(self.current_power) * dt

        # 循环老化近似：根据累计充放电量估算循环次数
        if abs(self.current_power) > 0.1:
            self.cycle_count += dt / 3600 * abs(self.current_power) / self.power_rating
        # 老化因子更新
        self.aging_factor_ESR = 1.0 + 0.0001 * self.cycle_count   # ESR随循环增加
        self.aging_factor_C = max(0.8, 1.0 - 0.00005 * self.cycle_count)  # 容量衰减不低于80%


class BESS(EnergyStorageDevice):  # 锂电池储能
    """锂电池模型，内阻随SOC变化"""
    def __init__(self):
        super().__init__("锂电池储能", power_rating=10.0, capacity=300.0,
                         charge_eff=0.92, discharge_eff=0.92,
                         op_cost=25, response_time=0.5, aging_coeff=3.5)
        self.voltage_nominal = 400      # 额定电压 (V)
        self.R0 = 0.002                  # 基准内阻 (Ω)

    def _calc_electrical(self, dt):
        # 内阻随SOC变化：在SOC=0.5时最小，偏离时增大
        soc_factor = 1.0 + 0.5 * (0.5 - self.soc)
        R = self.R0 * max(0.5, min(1.5, soc_factor))
        # 开路电压近似线性：Voc = 额定电压 * (0.8 + 0.4*SOC)
        Voc = self.voltage_nominal * (0.8 + 0.4 * self.soc)
        I, V = self._calc_current_voltage(self.current_power, Voc, R)
        self.current = I
        self.voltage = V


class CAES(EnergyStorageDevice):  # 压缩空气储能
    """压缩空气储能模型，考虑气体压力、燃料消耗"""
    def __init__(self):
        super().__init__("压缩空气储能", power_rating=5.0, capacity=162.0,
                         charge_eff=0.70, discharge_eff=0.65,
                         op_cost=5, response_time=1.0, aging_coeff=0.5, gas_price=3.6)
        self.voltage_nominal = 10000      # 额定电压 (V)
        self.R0 = 2.0                      # 内阻基准 (Ω)
        self.p0 = 1.0                      # 基准压力 (bar)
        self.k = 1.4                        # 气体绝热指数（空气）
        self.gas_consumption = deque(maxlen=10000)   # 燃气消耗记录

    def _calc_electrical(self, dt):
        # 压力随SOC线性变化（简化假设）
        p = self.p0 * (0.5 + 1.0 * self.soc)
        # 开路电压与压力比相关，等熵过程：Voc = V_nominal * (p/p0)^((k-1)/k)
        Voc = self.voltage_nominal * (p / self.p0) ** ((self.k - 1) / self.k)
        # 内阻随压力增大而减小（高压时导电性更好）
        R = self.R0 * (1.0 - 0.1 * (p - self.p0))
        I, V = self._calc_current_voltage(self.current_power, Voc, R)
        self.current = I
        self.voltage = V

    def _calc_fuel_cost(self, power, dt):
        """燃料成本：放电时消耗天然气，充电时可能无燃料消耗"""
        if power > 0:   # 放电时消耗燃料
            # 简化模型：每MWh放电消耗0.3 m³天然气
            gas_used = power * dt / 3600 * 0.3
            self.gas_consumption.append(gas_used)
            return gas_used * self.gas_price
        return 0.0


# ==================== 三、分层MPC控制器 ====================
class HierarchicalMPC:
    """
    分层模型预测控制（MPC）类。
    包含：
    - 上层经济调度（随机规划优化）
    - 下层实时平衡（小波分解频段分配）
    """
    def __init__(self, pred_horizon=10, control_horizon=5,
                 stochastic=True, n_scenarios=10, uncertainty_std=0.05):
        """
        初始化MPC控制器。

        参数:
        :param pred_horizon: 预测时域长度（步数），本简化版本未使用
        :param control_horizon: 控制时域长度，未使用
        :param stochastic: 是否启用随机规划（处理预测误差）
        :param n_scenarios: 随机场景数量
        :param uncertainty_std: 预测误差标准差（相对于当前净负荷的比例）
        """
        self.pred_horizon = pred_horizon
        self.control_horizon = control_horizon
        self.stochastic = stochastic
        self.n_scenarios = n_scenarios
        self.uncertainty_std = uncertainty_std
        self._buffer = None               # 用于小波分解的历史数据缓冲区
        self.last_scenarios = None         # 最近一次生成的随机场景，用于可视化
        self.buffer_length = 200            # 缓冲区长度，将在set_buffer_length中根据脉冲持续时间调整

    def set_buffer_length(self, dt, max_pulse_dur):
        """根据最大脉冲持续时间设置缓冲区长度，确保包含完整脉冲以便小波分解"""
        self.buffer_length = int(max_pulse_dur * 2 / dt) + 10

    def wavelet_decompose(self, signal, levels=5, wavelet='db4'):
        """
        小波包分解（或近似），将信号分解为低、中、高频三个分量。
        如果pywt可用，则使用小波包；否则使用移动平均近似。

        参数:
        :param signal: 输入信号数组
        :param levels: 分解层数
        :param wavelet: 小波基名称
        :return: 字典包含'low','mid','high'分量（与原始信号长度相同）
        """
        if not PYWT_AVAILABLE:
            # 移动平均近似：低频用较大窗口平均，中频用较小窗口平均，高频为残差
            low_window = max(1, len(signal) // 50)      # 低频窗口长度
            mid_window = max(1, len(signal) // 100)     # 中频窗口长度
            low = np.convolve(signal, np.ones(low_window)/low_window, mode='same')
            residual = signal - low
            mid = np.convolve(residual, np.ones(mid_window)/mid_window, mode='same')
            high = residual - mid
            return {'low': low, 'mid': mid, 'high': high}

        # 使用pywt进行小波包分解
        wp = pywt.WaveletPacket(data=signal, wavelet=wavelet,
                                mode='symmetric', maxlevel=levels)
        # 提取低频分量：所有 'a' 节点（例如 'aaaaa' 表示5层低频）
        low = getattr(wp['a'*levels], 'data', np.zeros_like(signal))
        # 提取高频分量：所有 'd' 节点（例如 'ddddd' 表示5层高频）
        high = getattr(wp['d'*levels], 'data', np.zeros_like(signal))
        # 如果分解后的长度与原始信号不一致，则插值到原始长度
        if len(low) != len(signal):
            low = np.interp(np.linspace(0,1,len(signal)), np.linspace(0,1,len(low)), low)
        if len(high) != len(signal):
            high = np.interp(np.linspace(0,1,len(signal)), np.linspace(0,1,len(high)), high)
        mid = signal - low - high      # 中频为剩余部分
        return {'low': low, 'mid': mid, 'high': high}

    def economic_dispatch(self, devices, net_load, time_idx, dt, eco_mode=False):
        """
        上层经济调度：基于MPC优化（简化版，仅优化当前时刻），考虑随机场景。

        参数:
        :param devices: 设备列表
        :param net_load: 当前净负荷 (MW)
        :param time_idx: 当前时间索引（用于确定电价时段）
        :param dt: 时间步长 (s)
        :param eco_mode: 是否启用环保模式（加入碳惩罚）
        :return: 各设备设定功率数组 (MW)
        """
        n = len(devices)
        if n == 0:
            return np.array([])

        # 根据当前时间确定电价（一天中的小时数）
        hour = (time_idx * dt) / 3600 % 24
        if 0 <= hour < 6 or 11 <= hour < 13:
            price = PRICE_VALLEY
        elif 14 <= hour < 22:
            price = PRICE_PEAK
        else:
            price = PRICE_FLAT

        track_penalty = 5.0   # 负荷跟踪惩罚系数（目标函数中二次项系数）

        # 生成随机场景（如果启用随机规划）
        if self.stochastic:
            np.random.seed(42 + time_idx)   # 固定随机种子，保证结果可重复
            base = max(abs(net_load), 0.1)   # 基准值，避免为零导致标准差为零
            sigma = self.uncertainty_std * base
            # 生成多个正态分布随机场景
            scenarios = net_load + np.random.normal(0, sigma, self.n_scenarios)
            scenarios = np.maximum(scenarios, 0)   # 净负荷不能为负（不考虑向电网馈电）
            weights = np.ones(self.n_scenarios) / self.n_scenarios   # 等权重
            self.last_scenarios = scenarios
        else:
            scenarios = np.array([net_load])
            weights = np.array([1.0])
            self.last_scenarios = None

        # 购电成本系数：将电价（元/kWh）转换为元/MW（即每MW功率在dt时间内的购电费用）
        # 公式：dt/3600 小时 * 电价 元/kWh * 1000 (kWh->MWh) = dt/3600 * price * 1000 元/MW
        const = dt / 3600 * price * 1000

        def objective(x):
            """
            目标函数：期望总成本 = 购电成本 + 储能成本 + 跟踪惩罚 + 碳惩罚
            x: 各设备功率数组
            """
            total_power = np.sum(x)
            # 储能成本（运维+老化+燃料）
            storage_cost = sum(
                dev._calc_op_cost(x[i], dt) +
                dev._calc_aging_cost(x[i], dt, dev.soc) +
                (dev._calc_fuel_cost(x[i], dt) if hasattr(dev, '_calc_fuel_cost') else 0.0)
                for i, dev in enumerate(devices)
            )
            expected_grid_cost = 0.0
            carbon_cost = 0.0
            # 对每个场景计算期望购电成本和碳成本
            for s, w in zip(scenarios, weights):
                grid_power = s - total_power
                if grid_power > 0:
                    expected_grid_cost += w * grid_power * const
                    if eco_mode:
                        # 碳惩罚：购电产生的碳排放量 * 碳价格
                        # 购电能量 (MWh) = grid_power * dt/3600，转换为kg CO2，再乘以价格
                        carbon_cost += w * (grid_power * dt / 3600 * 1000) * CARBON_FACTOR * CARBON_PRICE
            # 跟踪惩罚（二次型）确保总功率尽可能接近净负荷
            track_cost = track_penalty * (net_load - total_power) ** 2
            return expected_grid_cost + storage_cost + track_cost + carbon_cost

        # 设定各设备功率上下界（考虑SOC限制和额定功率）
        bounds, x0 = [], []
        for dev in devices:
            # 最大充电功率受限于可充电能量和额定功率
            max_charge = dev.get_charge_available() * 3600 / dt    # 最大允许充电功率 (MW)
            max_discharge = dev.get_discharge_available() * 3600 / dt
            p_min = -min(dev.power_rating, max_charge)
            p_max = min(dev.power_rating, max_discharge)
            bounds.append((p_min, p_max))
            x0.append(0.0)   # 初始猜测为0

        # 使用SLSQP求解器最小化目标函数
        res = minimize(objective, x0, method='SLSQP', bounds=bounds,
                       options={'maxiter': 500, 'ftol': 1e-9})
        if not res.success:
            # 若优化失败，返回零功率（安全策略）
            return np.zeros(n)
        return res.x

    def _allocate_frequency_power(self, devices, power_target, freq_type, dt):
        """
        根据频段类型分配功率给合适的设备（考虑边际成本和SOC限制）。

        参数:
        :param devices: 设备列表
        :param power_target: 目标功率 (MW)
        :param freq_type: 频段类型 'high','mid','low'
        :param dt: 时间步长
        :return: 分配字典 {设备名: 功率}
        """
        if abs(power_target) < 1e-6:
            return {}
        allocation = {}
        remaining = power_target

        if freq_type == 'high':
            # 高频设备优先级：超级电容器 > 飞轮储能 > 超导磁储能
            sc_devs = [d for d in devices if d.name == '超级电容器']
            if sc_devs:
                dev = sc_devs[0]
                # 计算该设备允许的功率范围
                charge_limit = -min(dev.power_rating, dev.get_charge_available() * 3600 / dt)
                discharge_limit = min(dev.power_rating, dev.get_discharge_available() * 3600 / dt)
                if remaining > 0:
                    assign = np.clip(remaining, 0, discharge_limit)
                else:
                    assign = np.clip(remaining, charge_limit, 0)
                if abs(assign) > 1e-6:
                    allocation[dev.name] = assign
                    remaining -= assign

            if abs(remaining) > 1e-6:
                fess_devs = [d for d in devices if d.name == '飞轮储能']
                if fess_devs:
                    dev = fess_devs[0]
                    charge_limit = -min(dev.power_rating, dev.get_charge_available() * 3600 / dt)
                    discharge_limit = min(dev.power_rating, dev.get_discharge_available() * 3600 / dt)
                    if remaining > 0:
                        assign = np.clip(remaining, 0, discharge_limit)
                    else:
                        assign = np.clip(remaining, charge_limit, 0)
                    if abs(assign) > 1e-6:
                        allocation[dev.name] = allocation.get(dev.name, 0.0) + assign
                        remaining -= assign

            if abs(remaining) > 1e-6:
                smes_devs = [d for d in devices if d.name == '超导磁储能']
                if smes_devs:
                    dev = smes_devs[0]
                    charge_limit = -min(dev.power_rating, dev.get_charge_available() * 3600 / dt)
                    discharge_limit = min(dev.power_rating, dev.get_discharge_available() * 3600 / dt)
                    if remaining > 0:
                        assign = np.clip(remaining, 0, discharge_limit)
                    else:
                        assign = np.clip(remaining, charge_limit, 0)
                    if abs(assign) > 1e-6:
                        allocation[dev.name] = allocation.get(dev.name, 0.0) + assign
                        remaining -= assign

        elif freq_type == 'mid':
            # 中频设备：飞轮储能、锂电池储能（按边际成本升序分配，低成本优先）
            eligible = [d for d in devices if d.name in ['飞轮储能', '锂电池储能']]
            eligible.sort(key=lambda d: d.marginal_cost)
            for dev in eligible:
                if abs(remaining) < 1e-6:
                    break
                charge_limit = -min(dev.power_rating, dev.get_charge_available() * 3600 / dt)
                discharge_limit = min(dev.power_rating, dev.get_discharge_available() * 3600 / dt)
                if remaining > 0:
                    assign = min(remaining, discharge_limit)
                else:
                    assign = max(remaining, charge_limit)
                if abs(assign) > 1e-6:
                    allocation[dev.name] = allocation.get(dev.name, 0.0) + assign
                    remaining -= assign

        else:  # low
            # 低频设备：压缩空气储能、锂电池储能（按边际成本升序）
            eligible = [d for d in devices if d.name in ['压缩空气储能', '锂电池储能']]
            eligible.sort(key=lambda d: d.marginal_cost)
            for dev in eligible:
                if abs(remaining) < 1e-6:
                    break
                charge_limit = -min(dev.power_rating, dev.get_charge_available() * 3600 / dt)
                discharge_limit = min(dev.power_rating, dev.get_discharge_available() * 3600 / dt)
                if remaining > 0:
                    assign = min(remaining, discharge_limit)
                else:
                    assign = max(remaining, charge_limit)
                if abs(assign) > 1e-6:
                    allocation[dev.name] = allocation.get(dev.name, 0.0) + assign
                    remaining -= assign

        return allocation

    def real_time_balance(self, devices, net_load, dt):
        """
        实时平衡：通过小波分解将净负荷分解为高频、中频、低频，并分别分配给对应设备。

        参数:
        :param devices: 设备列表
        :param net_load: 当前净负荷 (MW)
        :param dt: 时间步长
        :return: 分配字典 {设备名: 功率}
        """
        # 维护历史缓冲区
        if self._buffer is None:
            self._buffer = np.ones(self.buffer_length) * net_load
        else:
            # 滚动更新：将缓冲区所有元素前移，并在末尾插入新值
            self._buffer = np.roll(self._buffer, -1)
            self._buffer[-1] = net_load

        # 对缓冲区信号进行小波分解
        comp = self.wavelet_decompose(self._buffer, levels=5)
        # 取最新时刻的分解值作为当前频段功率需求
        high = comp['high'][-1] if len(comp['high']) else 0
        mid = comp['mid'][-1] if len(comp['mid']) else 0
        low = comp['low'][-1] if len(comp['low']) else 0

        alloc = {}
        # 分别分配各频段功率
        alloc_high = self._allocate_frequency_power(devices, high, 'high', dt)
        for k, v in alloc_high.items():
            alloc[k] = alloc.get(k, 0.0) + v

        alloc_mid = self._allocate_frequency_power(devices, mid, 'mid', dt)
        for k, v in alloc_mid.items():
            alloc[k] = alloc.get(k, 0.0) + v

        alloc_low = self._allocate_frequency_power(devices, low, 'low', dt)
        for k, v in alloc_low.items():
            alloc[k] = alloc.get(k, 0.0) + v

        return alloc


# ==================== 四、混合储能系统主类 ====================
class HybridEnergyStorageSystem:
    """混合储能系统主控类，协调设备、MPC控制器、仿真运行与监控"""
    def __init__(self):
        # 设备规格字典，用于根据名称创建设备实例
        self.device_specs = {
            '飞轮储能': {'class': FESS},
            '超导磁储能': {'class': SMES},
            '超级电容器': {'class': SC},
            '锂电池储能': {'class': BESS},
            '压缩空气储能': {'class': CAES}
        }
        self.devices = []                       # 当前选中的设备实例列表
        self.mpc = HierarchicalMPC(stochastic=True, n_scenarios=10, uncertainty_std=0.05)

    def _clip_power_to_limits(self, power, dev, dt):
        """根据SOC限制和额定功率对功率进行限幅"""
        max_charge = dev.get_charge_available() * 3600 / dt
        max_discharge = dev.get_discharge_available() * 3600 / dt
        p_min = -min(dev.power_rating, max_charge)
        p_max = min(dev.power_rating, max_discharge)
        return np.clip(power, p_min, p_max)

    def _adjust_power_to_track(self, set_powers, devices, target, dt):
        """
        二次调整：使总功率精确等于目标值（考虑各设备剩余调节能力）。

        参数:
        :param set_powers: 当前设定功率数组
        :param devices: 设备列表
        :param target: 目标总功率 (净负荷)
        :param dt: 时间步长
        :return: 调整后的功率数组
        """
        current_total = np.sum(set_powers)
        diff = target - current_total
        if abs(diff) < TRACK_TOLERANCE:
            return set_powers

        # 计算每个设备可向上/向下调节的容量
        capacities = []
        for i, dev in enumerate(devices):
            max_charge = dev.get_charge_available() * 3600 / dt
            max_discharge = dev.get_discharge_available() * 3600 / dt
            p_min = -min(dev.power_rating, max_charge)
            p_max = min(dev.power_rating, max_discharge)
            if diff > 0:   # 需要增加放电或减少充电
                room = p_max - set_powers[i]
            else:           # 需要减少放电或增加充电
                room = set_powers[i] - p_min
            capacities.append(max(0, room))

        total_capacity = np.sum(capacities)
        if total_capacity < abs(diff):
            # 调节能力不足，按最大能力调节（scale=1）
            scale = 1.0
        else:
            scale = abs(diff) / total_capacity

        adjusted = set_powers.copy()
        for i in range(len(devices)):
            if capacities[i] > 0:
                if diff > 0:
                    adjusted[i] += capacities[i] * scale
                else:
                    adjusted[i] -= capacities[i] * scale
        return adjusted

    def _enforce_no_sellback(self, set_powers, demand):
        """
        强制不允许向电网反送电：如果总出力超过需求，削减放电功率或增加充电。
        参数:
        :param set_powers: 当前设定功率数组
        :param demand: 净负荷需求
        :return: 调整后的功率数组（确保总功率 <= demand）
        """
        total = np.sum(set_powers)
        if total <= demand + 1e-6:
            return set_powers
        excess = total - demand
        # 优先削减放电设备（功率>0）
        discharge_indices = [i for i, p in enumerate(set_powers) if p > 0]
        if discharge_indices:
            discharge_powers = np.array([set_powers[i] for i in discharge_indices])
            total_discharge = np.sum(discharge_powers)
            reduction_factors = discharge_powers / total_discharge
            reductions = reduction_factors * excess
            for idx, red in zip(discharge_indices, reductions):
                set_powers[idx] = max(0, set_powers[idx] - red)
        else:
            # 如果没有放电设备，则增加充电（功率<0）
            charge_indices = [i for i, p in enumerate(set_powers) if p < 0]
            if charge_indices:
                charge_powers = np.array([set_powers[i] for i in charge_indices])
                abs_charges = np.abs(charge_powers)
                total_abs = np.sum(abs_charges)
                if total_abs > 0:
                    reductions = excess * (abs_charges / total_abs)
                    for idx, red in zip(charge_indices, reductions):
                        set_powers[idx] = min(0, set_powers[idx] + red)  # 增加充电（负值绝对值增大）
        return set_powers

    def _update_trend_graph(self, dev_name, power):
        """更新实时趋势图（在监控窗口中绘制小功率曲线）"""
        max_points = 50
        if dev_name not in self.trend_data:
            self.trend_data[dev_name] = deque(maxlen=max_points)
        data = self.trend_data[dev_name]
        data.append(power)

        if dev_name in self.trend_canvases:
            canvas = self.trend_canvases[dev_name]
            canvas.delete("trend")
            if len(data) < 2:
                return
            w = canvas.winfo_width()
            h = canvas.winfo_height()
            if w <= 1 or h <= 1:
                return
            points = []
            for i, val in enumerate(data):
                x = (i / (max_points - 1)) * w
                # 找到对应设备获取额定功率，用于归一化
                dev = next(d for d in self.devices if d.name == dev_name)
                rating = dev.power_rating
                # 将功率映射到画布高度：中心线为0，上下幅度为额定功率
                y = h/2 - (val / rating) * (h/2 - 5)   # 减去5留边距
                points.append((x, y))
            if len(points) > 1:
                canvas.create_line(points, fill=COLOR_ACCENT, width=2, tags="trend")
            # 绘制零线
            canvas.create_line(0, h/2, w, h/2, fill=COLOR_TEXT_SECONDARY, dash=(2,2), tags="trend")

    def simulate(self, pulse_amp, pulse_dur, plant_power=10.0, sim_time=700, dt=0.1, econ_interval=1, eco_mode=False):
        """
        执行仿真。

        参数:
        :param pulse_amp: 脉冲幅值 (MW)
        :param pulse_dur: 脉冲持续时间 (s)
        :param plant_power: 电厂恒发功率 (MW)
        :param sim_time: 总仿真时间 (s)
        :param dt: 仿真步长 (s)
        :param econ_interval: 经济调度间隔（每隔多少步执行一次）
        :param eco_mode: 是否开启环保模式
        :return: 仿真结果字典
        """
        np.random.seed(42)          # 固定随机种子，保证结果可重复
        time = np.arange(0, sim_time, dt)
        n = len(time)

        # 基础负荷 = 电厂功率（假设负荷等于电厂功率加上脉冲）
        load = plant_power * np.ones_like(time)
        # 叠加脉冲
        start = int(200.0 / dt)      # 脉冲起始索引（200秒开始）
        end = int((200.0 + pulse_dur) / dt)
        load[start:end] += pulse_amp

        plant = plant_power * np.ones_like(time)
        net_load = load - plant      # 净负荷 = 负荷 - 电厂出力，即需要储能平衡的部分

        # 设置MPC缓冲区长度（至少能覆盖脉冲）
        self.mpc.set_buffer_length(dt, max(pulse_dur, 20))

        # 初始化结果字典
        res = {
            'time': time, 'load': load, 'plant': plant, 'net_load': net_load,
            'smoothed_load': np.zeros_like(time),    # 平滑后负荷 = 电厂 + 储能总功率
            'total_ess': np.zeros_like(time),         # 储能总功率
            'grid_power': np.zeros_like(time),        # 电网购电功率（净负荷 - 储能总功率，但非负）
            'grid_purchase': np.zeros_like(time),     # 购电费用（每个时间步）
            'device_power': {d.name: np.zeros_like(time) for d in self.devices},
            'device_soc': {d.name: np.zeros_like(time) for d in self.devices},
            'device_energy': {d.name: np.zeros_like(time) for d in self.devices},
            'device_cost': {d.name: np.zeros_like(time) for d in self.devices},
            'device_op_cost': {d.name: np.zeros_like(time) for d in self.devices},
            'device_aging_cost': {d.name: np.zeros_like(time) for d in self.devices},
            'device_fuel_cost': {d.name: np.zeros_like(time) for d in self.devices},
            'device_current': {d.name: np.zeros_like(time) for d in self.devices},
            'device_voltage': {d.name: np.zeros_like(time) for d in self.devices},
            'pulse_amp': pulse_amp, 'pulse_dur': pulse_dur,
            'eco_mode': eco_mode
        }
        self.sim_res = res

        # 成本累计变量
        total_purchase = 0.0
        total_op = 0.0
        total_aging = 0.0
        total_fuel = 0.0
        total_carbon_penalty = 0.0

        last_econ_powers = np.zeros(len(self.devices))   # 上一次经济调度结果
        econ_cnt = 0

        self._update_monitor(init=True)   # 初始化监控窗口
        self.paused = False
        self.stop_flag = False

        # 主仿真循环
        for t in range(n):
            # 处理暂停/停止（用于GUI交互）
            while self.paused and not self.stop_flag:
                self.root.update()
                plt.pause(0.1)
            if self.stop_flag:
                break

            demand = net_load[t]

            # 每隔econ_interval步执行一次经济调度
            if econ_cnt == 0:
                last_econ_powers = self.mpc.economic_dispatch(self.devices, demand, t, dt, eco_mode=eco_mode)
            econ_cnt = (econ_cnt + 1) % econ_interval

            # 实时平衡：得到频段分配功率
            alloc = self.mpc.real_time_balance(self.devices, demand, dt)

            # 组合功率 = 经济调度功率 + 实时平衡功率
            set_powers = last_econ_powers.copy()
            for i, dev in enumerate(self.devices):
                if dev.name in alloc:
                    set_powers[i] += alloc[dev.name]

            # 功率限幅（考虑SOC和额定功率）
            for i, dev in enumerate(self.devices):
                set_powers[i] = self._clip_power_to_limits(set_powers[i], dev, dt)

            # 二次调整使总功率精确跟踪净负荷
            set_powers = self._adjust_power_to_track(set_powers, self.devices, demand, dt)

            # 防止反送电
            set_powers = self._enforce_no_sellback(set_powers, demand)

            total_ess = 0.0
            # 更新各设备状态
            for i, dev in enumerate(self.devices):
                set_power = set_powers[i]
                soc, cost, energy = dev.update_state(set_power, dt)
                total_op += dev.history_op_cost[-1]
                total_aging += dev.history_aging_cost[-1]
                total_fuel += dev.history_fuel_cost[-1]

                # 记录结果
                res['device_power'][dev.name][t] = dev.current_power
                res['device_soc'][dev.name][t] = soc
                res['device_energy'][dev.name][t] = energy
                res['device_cost'][dev.name][t] = cost
                res['device_op_cost'][dev.name][t] = dev.history_op_cost[-1]
                res['device_aging_cost'][dev.name][t] = dev.history_aging_cost[-1]
                res['device_fuel_cost'][dev.name][t] = dev.history_fuel_cost[-1]
                res['device_current'][dev.name][t] = dev.current
                res['device_voltage'][dev.name][t] = dev.voltage
                total_ess += dev.current_power
                self._update_trend_graph(dev.name, dev.current_power)

            res['total_ess'][t] = total_ess
            res['smoothed_load'][t] = plant[t] + total_ess

            # 电网购电功率（不可为负，即不允许储能向电网馈电）
            res['grid_power'][t] = max(0, demand - total_ess)

            # 购电费用计算
            hour = (t * dt) / 3600 % 24
            if 0 <= hour < 6 or 11 <= hour < 13:
                price = PRICE_VALLEY
            elif 14 <= hour < 22:
                price = PRICE_PEAK
            else:
                price = PRICE_FLAT

            if res['grid_power'][t] > 0:
                purchase = res['grid_power'][t] * dt / 3600 * price * 1000
                res['grid_purchase'][t] = purchase
                total_purchase += purchase

            if eco_mode:
                carbon_penalty = (res['grid_power'][t] * dt / 3600 * 1000) * CARBON_FACTOR * CARBON_PRICE
                total_carbon_penalty += carbon_penalty

            # 更新累计总成本和当前电网功率，供监控显示
            self.current_total_cost = total_purchase + total_op + total_aging + total_fuel + total_carbon_penalty
            self.current_grid_power = res['grid_power'][t]
            self._update_monitor(self.current_total_cost, self.current_grid_power)
            self.root.update()

        # 保存最终成本
        self.total_purchase = total_purchase
        self.total_op = total_op
        self.total_aging = total_aging
        self.total_fuel = total_fuel
        self.total_carbon_penalty = total_carbon_penalty
        self.total_cost = self.current_total_cost
        self.net_cost = self.total_cost   # 此处无补贴，净成本等于总成本

        return res

    def run_sim(self):
        """启动仿真"""
        if self.sim_running:
            messagebox.showinfo("提示", "仿真正在进行中，请稍候...")
            return

        amp = self.pulse_amp.get()
        dur = self.pulse_dur.get()
        plant_power = self.plant_power.get()
        if amp <= 0 or dur <= 0 or plant_power <= 0:
            messagebox.showerror("错误", "脉冲幅值、持续时间和电厂功率必须为正数！")
            return

        selected = [name for name, var in self.dev_vars.items() if var.get()]
        if not selected:
            messagebox.showerror("错误", "至少选择一个储能设备！")
            return

        self.start_btn.config(state='disabled')
        self.status.config(text="仿真进行中... 正在消除脉冲...", fg=COLOR_WARNING)
        self.progress.start(10)
        self.root.update()
        self.sim_running = True

        try:
            hess = HybridEnergyStorageSystem()
            hess.root = self.root
            hess.mpc.stochastic = self.use_stochastic.get()
            hess.mpc.n_scenarios = self.n_scenarios.get()
            hess.mpc.uncertainty_std = self.uncertainty_std.get()
            hess.create_devices(selected)
            results = hess.simulate(amp, dur, plant_power=plant_power,
                                    sim_time=700, econ_interval=1, eco_mode=self.eco_mode.get())

            self.status.config(text="仿真完成！", fg=COLOR_SUCCESS)
            self.start_btn.config(state='normal')
            self.progress.stop()
            self.sim_running = False
            self.show_results(hess, results, amp, dur, selected)
        except Exception as e:
            messagebox.showerror("错误", f"仿真异常：{str(e)}")
            self.status.config(text="仿真失败", fg=COLOR_DANGER)
            self.start_btn.config(state='normal')
            self.progress.stop()
            self.sim_running = False

    def show_results(self, hess, res, amp, dur, selected):
        """显示仿真结果窗口（包含详细文本数据）"""
        # 提取累计成本等
        total_purchase = hess.total_purchase
        total_op = hess.total_op
        total_aging = hess.total_aging
        total_fuel = hess.total_fuel
        total_carbon = hess.total_carbon_penalty
        total_cost = hess.total_cost
        net_cost = hess.net_cost

        dt = res['time'][1] - res['time'][0]

        # 计算无储能时的总购电量（假设所有净负荷都由电网提供）
        total_grid_without_storage = np.sum(np.maximum(res['net_load'], 0)) * dt / 3600
        total_grid_with_storage = np.sum(res['grid_power']) * dt / 3600
        grid_reduction = total_grid_without_storage - total_grid_with_storage
        carbon_reduction = grid_reduction * 1000 * CARBON_FACTOR

        # 各设备充放电统计
        dev_charge = {}
        dev_discharge = {}
        dev_avg_p = {}
        dev_max_p = {}
        for dev in hess.devices:
            p = res['device_power'][dev.name]
            dev_charge[dev.name] = -np.sum(p[p < 0]) * dt / 3600
            dev_discharge[dev.name] = np.sum(p[p > 0]) * dt / 3600
            dev_avg_p[dev.name] = np.mean(np.abs(p))
            dev_max_p[dev.name] = np.max(np.abs(p))

        # 脉冲期间平均功率
        mask = (res['time'] >= 200) & (res['time'] < 200 + dur)
        pulse_power = {}
        for dev in hess.devices:
            power_in_pulse = res['device_power'][dev.name][mask]
            positive_power = power_in_pulse[power_in_pulse > 0]
            if len(positive_power) > 0:
                avg = np.mean(positive_power)
                if avg > 0.01:
                    pulse_power[dev.name] = avg

        # 平滑评分
        rmse = np.sqrt(np.mean((res['load'] - res['smoothed_load'])**2))
        pulse_amp_val = res['pulse_amp']
        smooth_score = max(0, min(100, 100 * (1 - rmse / (pulse_amp_val + 1e-6))))
        max_dev = np.max(np.abs(res['load'] - res['smoothed_load']))

        # 创建新窗口
        win = tk.Toplevel()
        win.title("脉冲平滑仿真详细结果")

        tk.Label(row1, text=f"{smooth_score:.1f} 分 (满分100)", bg=COLOR_BG, font=('微软雅黑',12,'bold'),

    def export_report(self, hess, res, selected, amp, dur, total_cost, carbon_reduction, smooth_score):
        """导出文本报告到文件"""
        from tkinter import filedialog
        filename = filedialog.asksaveasfilename(defaultextension=".txt",
                                                filetypes=[("Text files", "*.txt")])
        if not filename:
            return
        with open(filename, 'w', encoding='utf-8') as f:
            f.write("混合储能系统脉冲平滑仿真报告\n")
            f.write("="*50 + "\n")
            f.write(f"仿真时间: {res['time'][-1]:.1f} 秒\n")
            f.write(f"脉冲幅值: {amp:.2f} MW\n")
            f.write(f"脉冲持续时间: {dur:.2f} s\n")
            f.write(f"总成本: {total_cost:.2f} 元\n")
            f.write(f"碳减排量: {carbon_reduction:.2f} kg CO₂\n")
            f.write(f"平滑评分: {smooth_score:.1f}\n")
            f.write("\n设备清单:\n")
            for name in selected:
                f.write(f"  {name}\n")
        messagebox.showinfo("导出成功", f"报告已保存至 {filename}")

    def plot_scenarios(self, scenarios):
        """绘制随机规划生成的场景（当前时刻）"""
        plt.figure(figsize=(10, 6))
        x = np.arange(len(scenarios))
        for i, s in enumerate(scenarios):
            plt.plot(x, [s]*len(x), 'b-', alpha=0.2, linewidth=0.5)  # 每个场景用水平线表示
        mean_val = np.mean(scenarios)
        std_val = np.std(scenarios)
        plt.axhline(mean_val, color='r', linestyle='--', label=f'均值: {mean_val:.2f} MW')
        plt.fill_between(x, mean_val - std_val, mean_val + std_val, color='r', alpha=0.2, label='±1σ 区间')
        plt.xlabel('场景编号')
        plt.ylabel('净负荷 (MW)')
        plt.title('随机规划生成的净负荷预测场景（当前时刻）')
        plt.legend()
        plt.grid(True, alpha=0.3)
        plt.tight_layout()
        plt.show()

    def plot_all(self, hess, res):
        """
        将结果绘制到多个独立的图形窗口中，每个窗口尺寸合适，彻底避免文字重叠。

        4. 所有设备电压-时间曲线（多子图）
        5. 所有设备电流-时间曲线（多子图）
        6. 所有设备储存能量-时间曲线（多子图）
        """

            plt.rcParams['font.sans-serif'] = ['SimHei']
            plt.rcParams['axes.unicode_minus'] = False

            devices = hess.devices

            fig1 = plt.figure(figsize=(12, 10))

            # 子图1：负荷跟踪效果
            ax1 = fig1.add_subplot(2, 2, 1)

            ax1.plot(time, res['smoothed_load'], 'b--', lw=1.5, label='电厂+储能总出力')
            ax1.plot(time, res['plant'], 'g-', lw=1, label=f'电厂出力({res["plant"][0]:.0f}MW)')
            ax1.fill_between(time, res['load'], res['smoothed_load'], alpha=0.2, color='gray')
            ax1.set_xlabel('时间 (秒)', fontsize=9); ax1.set_ylabel('功率 (MW)', fontsize=9)
            ax1.set_title('负荷跟踪效果', fontsize=11); ax1.legend(fontsize=8)
            ax1.grid(True, alpha=0.3)

            # 子图2：储能总功率与电网购电
            ax2 = fig1.add_subplot(2, 2, 2)
            ax2.plot(time, res['total_ess'], 'purple', lw=1.5, label='储能总功率')
            ax2.plot(time, res['grid_power'], 'orange', lw=1, label='电网购电功率')
            ax2.axhline(0, color='black', ls='--', alpha=0.5)
            ax2.set_xlabel('时间 (秒)', fontsize=9); ax2.set_ylabel('功率 (MW)', fontsize=9)
            ax2.set_title('储能总功率与电网购电', fontsize=11); ax2.legend(fontsize=8)
            ax2.grid(True, alpha=0.3)

            # 子图3：脉冲局部放大
            ax3 = fig1.add_subplot(2, 2, 3)
            mask_local = (time >= 198) & (time <= 202 + res['pulse_dur'])
            ax3.plot(time[mask_local], res['load'][mask_local], 'r-', lw=1.5, label='原始负荷')
            ax3.plot(time[mask_local], res['smoothed_load'][mask_local], 'b--', lw=1.5, label='电厂+储能')
            ax3.set_xlabel('时间 (秒)', fontsize=9); ax3.set_ylabel('功率 (MW)', fontsize=9)
            ax3.set_title('脉冲局部放大', fontsize=11); ax3.legend(fontsize=8)
            ax3.grid(True, alpha=0.3)

            # 子图4：购电费用
            ax4 = fig1.add_subplot(2, 2, 4)
            ax4.plot(time, res['grid_purchase'], 'r-', lw=1, label='购电支出')
            ax4.set_xlabel('时间 (秒)', fontsize=9); ax4.set_ylabel('金额 (元)', fontsize=9)
            ax4.set_title('电网购电费用', fontsize=11); ax4.legend(fontsize=8)
            ax4.grid(True, alpha=0.3)

            plt.tight_layout(pad=3.0)

            # ---------- 窗口2：设备状态概览 ----------
            fig2 = plt.figure(figsize=(14, 10))
            fig2.suptitle('设备状态概览', fontsize=14, fontweight='bold')

            # 子图1：各设备电流变化
            ax21 = fig2.add_subplot(2, 2, 1)
            for dev in devices:
                ax21.plot(time, res['device_current'][dev.name], label=dev.name, linewidth=1)
            ax21.set_xlabel('时间 (秒)', fontsize=9); ax21.set_ylabel('电流 (A)', fontsize=9)
            ax21.set_title('各设备电流变化', fontsize=11); ax21.legend(fontsize=7, ncol=2)
            ax21.grid(True, alpha=0.3)

            # 子图2：各设备电压变化
            ax22 = fig2.add_subplot(2, 2, 2)
            for dev in devices:
                ax22.plot(time, res['device_voltage'][dev.name], label=dev.name, linewidth=1)
            ax22.set_xlabel('时间 (秒)', fontsize=9); ax22.set_ylabel('电压 (V)', fontsize=9)
            ax22.set_title('各设备电压变化', fontsize=11); ax22.legend(fontsize=7, ncol=2)
            ax22.grid(True, alpha=0.3)

            # 子图3：各设备SOC变化
            ax23 = fig2.add_subplot(2, 2, 3)
            for dev in devices:
                ax23.plot(time, res['device_soc'][dev.name], label=dev.name, linewidth=1)
            ax23.set_xlabel('时间 (秒)', fontsize=9); ax23.set_ylabel('SOC', fontsize=9)
            ax23.set_title('各设备荷电状态', fontsize=11); ax23.legend(fontsize=7, ncol=2)
            ax23.grid(True, alpha=0.3)

            # 子图4：实时可充放电能量
            ax24 = fig2.add_subplot(2, 2, 4)
            for dev in devices:
                soc_arr = res['device_soc'][dev.name]
                chargeable = (SOC_MAX - soc_arr) * dev.capacity
                dischargeable = (soc_arr - SOC_MIN) * dev.capacity
                # 主要设备用实线，其他设备用灰色虚线
                if dev.name in ['锂电池储能', '压缩空气储能']:
                    ax24.plot(time, chargeable, '--', lw=0.8, label=f'{dev.name}可充')
                    ax24.plot(time, dischargeable, '-', lw=0.8, label=f'{dev.name}可放')
                else:
                    ax24.plot(time, chargeable, '--', lw=0.8, color='gray')
                    ax24.plot(time, dischargeable, '-', lw=0.8, color='gray')
            ax24.set_xlabel('时间 (秒)', fontsize=9); ax24.set_ylabel('能量 (MWh)', fontsize=9)
            ax24.set_title('实时可充放电能量', fontsize=11); ax24.legend(fontsize=7, ncol=2)
            ax24.grid(True, alpha=0.3)

            plt.tight_layout(pad=3.0)

            # ---------- 窗口3：功率分配与成本 ----------
            fig3 = plt.figure(figsize=(14, 6))
            fig3.suptitle('功率分配与成本', fontsize=14, fontweight='bold')

            # 子图1：脉冲期间功率分配饼图（图例放在外面）
            ax31 = fig3.add_subplot(1, 3, 1)
            mask = (time >= 200) & (time < 200 + res['pulse_dur'])
            pulse_power = {}
            for dev in devices:
                p = res['device_power'][dev.name][mask]
                pos = p[p > 0]
                if len(pos) > 0:
                    avg = np.mean(pos)
                    if avg > 0.05:
                        pulse_power[dev.name] = avg
            if pulse_power:
                labels = list(pulse_power.keys())
                sizes = list(pulse_power.values())
                wedges, texts, autotexts = ax31.pie(sizes, autopct='%1.1f%%',
                                                     startangle=90,
                                                     textprops={'fontsize': 8},
                                                     pctdistance=0.8)
                ax31.legend(wedges, labels, loc='center left', bbox_to_anchor=(1, 0, 0.5, 1), fontsize=8)
                ax31.set_title('脉冲期间功率分配', fontsize=11)
                ax31.axis('equal')
            else:
                ax31.text(0.5, 0.5, '无脉冲数据', ha='center', va='center', fontsize=10)
                ax31.set_title('脉冲期间功率分配', fontsize=11)

            # 子图2：储能出力堆叠
            ax32 = fig3.add_subplot(1, 3, 2)
            pos_data = [np.maximum(res['device_power'][dev.name], 0) for dev in devices]
            neg_data = [np.minimum(res['device_power'][dev.name], 0) for dev in devices]
            labels = [dev.name for dev in devices]
            ax32.stackplot(time, *pos_data, labels=labels, alpha=0.7, baseline='zero')
            ax32.stackplot(time, *neg_data, alpha=0.7, baseline='zero')
            ax32.set_xlabel('时间 (秒)', fontsize=9); ax32.set_ylabel('功率 (MW)', fontsize=9)
            ax32.set_title('储能出力堆叠', fontsize=11)
            ax32.legend(loc='upper right', fontsize=7, ncol=1)
            ax32.grid(True, alpha=0.3)

            # 子图3：各设备边际成本对比
            ax33 = fig3.add_subplot(1, 3, 3)
            names = [dev.name for dev in devices]
            costs = [dev.marginal_cost for dev in devices]
            bars = ax33.bar(names, costs, color='skyblue')
            ax33.set_xlabel('设备', fontsize=9); ax33.set_ylabel('边际成本 (元/MW/h)', fontsize=9)
            ax33.set_title('各设备边际成本对比', fontsize=11)
            ax33.tick_params(axis='x', rotation=30, labelsize=8)
            ax33.tick_params(axis='y', labelsize=8)
            for bar in bars:
                height = bar.get_height()
                ax33.text(bar.get_x() + bar.get_width()/2., height + 0.5,
                         f'{height:.1f}', ha='center', va='bottom', fontsize=7)
            plt.tight_layout(pad=3.0)

            # ---------- 窗口4：所有设备电压-时间曲线 ----------
            if n_dev > 0:
                n_cols = 2
                n_rows = (n_dev + n_cols - 1) // n_cols
                fig4 = plt.figure(figsize=(12, 4 * n_rows))
                fig4.suptitle('各设备电压-时间曲线', fontsize=14, fontweight='bold')
                for idx, dev in enumerate(devices):
                    ax = fig4.add_subplot(n_rows, n_cols, idx + 1)
                    ax.plot(time, res['device_voltage'][dev.name], color='C0', linewidth=1)
                    ax.set_xlabel('时间 (秒)', fontsize=8); ax.set_ylabel('电压 (V)', fontsize=8)
                    ax.set_title(dev.name, fontsize=10)
                    ax.grid(True, alpha=0.3)
                    ax.tick_params(labelsize=7)
                plt.tight_layout(pad=2.5)

            # ---------- 窗口5：所有设备电流-时间曲线 ----------
            if n_dev > 0:
                n_cols = 2
                n_rows = (n_dev + n_cols - 1) // n_cols
                fig5 = plt.figure(figsize=(12, 4 * n_rows))
                fig5.suptitle('各设备电流-时间曲线', fontsize=14, fontweight='bold')
                for idx, dev in enumerate(devices):
                    ax = fig5.add_subplot(n_rows, n_cols, idx + 1)
                    ax.plot(time, res['device_current'][dev.name], color='C1', linewidth=1)
                    ax.set_xlabel('时间 (秒)', fontsize=8); ax.set_ylabel('电流 (A)', fontsize=8)
                    ax.set_title(dev.name, fontsize=10)
                    ax.grid(True, alpha=0.3)
                    ax.tick_params(labelsize=7)
                plt.tight_layout(pad=2.5)

            # ---------- 窗口6：所有设备储存能量-时间曲线 ----------
            if n_dev > 0:

                n_rows = (n_dev + n_cols - 1) // n_cols
                fig6 = plt.figure(figsize=(12, 4 * n_rows))
                fig6.suptitle('各设备储存能量-时间曲线', fontsize=14, fontweight='bold')

                    ax.tick_params(labelsize=7)
                plt.tight_layout(pad=2.5)
